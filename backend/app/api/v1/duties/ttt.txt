-- Table: public.duties

-- DROP TABLE IF EXISTS public.duties;

CREATE TABLE IF NOT EXISTS public.duties
(
    id integer NOT NULL DEFAULT nextval('duties_id_seq'::regclass),
    duty_type character varying COLLATE pg_catalog."default" NOT NULL,
    room_number integer NOT NULL,
    floor integer NOT NULL,
    date_due timestamp without time zone NOT NULL,
    assigned_by_id integer NOT NULL,
    status character varying COLLATE pg_catalog."default" NOT NULL,
    notes text COLLATE pg_catalog."default",
    created_at timestamp with time zone DEFAULT now(),
    updated_at timestamp with time zone,
    date_assigned timestamp without time zone DEFAULT now(),
    CONSTRAINT duties_pkey PRIMARY KEY (id),
    CONSTRAINT duties_assigned_by_id_fkey FOREIGN KEY (assigned_by_id)
        REFERENCES public.users (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT duties_status_check CHECK (status::text = ANY (ARRAY['pending'::character varying, 'verified'::character varying]::text[]))
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.duties
    OWNER to postgres;
-- Index: ix_duties_id

-- DROP INDEX IF EXISTS public.ix_duties_id;

CREATE INDEX IF NOT EXISTS ix_duties_id
    ON public.duties USING btree
    (id ASC NULLS LAST)
    WITH (fillfactor=100, deduplicate_items=True)
    TABLESPACE pg_default;

CREATE TABLE IF NOT EXISTS public.duty_reports
(
    id integer NOT NULL DEFAULT nextval('duty_reports_id_seq'::regclass),
    duty_id integer NOT NULL,
    student_id integer NOT NULL,
    description text COLLATE pg_catalog."default" NOT NULL,
    submitted_at timestamp without time zone DEFAULT now(),
    status character varying(20) COLLATE pg_catalog."default",
    reviewed_at timestamp without time zone,
    reviewed_by integer,
    review_notes text COLLATE pg_catalog."default",
    CONSTRAINT duty_reports_pkey PRIMARY KEY (id),
    CONSTRAINT duty_reports_duty_id_fkey FOREIGN KEY (duty_id)
        REFERENCES public.duties (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT duty_reports_reviewed_by_fkey FOREIGN KEY (reviewed_by)
        REFERENCES public.users (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT duty_reports_student_id_fkey FOREIGN KEY (student_id)
        REFERENCES public.users (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT duty_reports_review_check CHECK (status::text <> 'verified'::text OR reviewed_at IS NOT NULL AND reviewed_by IS NOT NULL),
    CONSTRAINT duty_reports_status_check CHECK (status::text = ANY (ARRAY['submitted'::character varying, 'reviewing'::character varying, 'verified'::character varying]::text[]))
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.duty_reports
    OWNER to postgres;

CREATE TABLE IF NOT EXISTS public.users
(
    id integer NOT NULL DEFAULT nextval('users_id_seq'::regclass),
    role_id integer NOT NULL DEFAULT 3,
    name character varying(100) COLLATE pg_catalog."default" NOT NULL,
    surname character varying(100) COLLATE pg_catalog."default" NOT NULL,
    password_hash character varying(255) COLLATE pg_catalog."default" NOT NULL,
    phone character varying(20) COLLATE pg_catalog."default" NOT NULL,
    email character varying(100) COLLATE pg_catalog."default",
    user_group character varying(10) COLLATE pg_catalog."default",
    n_room integer,
    is_active boolean DEFAULT true,
    created_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT users_pkey PRIMARY KEY (id),
    CONSTRAINT users_email_key UNIQUE (email),
    CONSTRAINT users_phone_key UNIQUE (phone),
    CONSTRAINT users_role_id_fkey FOREIGN KEY (role_id)
        REFERENCES public.roles (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

    OWNER to postgres;
