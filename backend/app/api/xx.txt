-- Table: public.duties

-- DROP TABLE IF EXISTS public.duties;

CREATE TABLE IF NOT EXISTS public.duties
(
    id integer NOT NULL DEFAULT nextval('duties_id_seq'::regclass),
    duty_type character varying COLLATE pg_catalog."default" NOT NULL,
    room_number integer NOT NULL,
    floor integer NOT NULL,
    date_assigned timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    date_due timestamp without time zone NOT NULL,
    assigned_by_id integer NOT NULL,
    status character varying COLLATE pg_catalog."default" NOT NULL,
    notes text COLLATE pg_catalog."default",
    created_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamp without time zone,
    CONSTRAINT duties_pkey PRIMARY KEY (id),
    CONSTRAINT duties_assigned_by_id_fkey FOREIGN KEY (assigned_by_id)
        REFERENCES public.users (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT duties_status_check CHECK (status::text = ANY (ARRAY['assigned'::character varying, 'confirmed'::character varying, 'in_progress'::character varying, 'completed'::character varying, 'verified'::character varying, 'reported'::character varying, 'pending'::character varying, 'waiting'::character varying, 'submitted'::character varying]::text[]))
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.duties
    OWNER to postgres;

        ON DELETE NO ACTION
